class Observer{
    constructor() {
        this.observer = {};
    }
    listener(type, fn) {
        if (!type ||  typeof fn !== 'function') {
            return;
        }
        if (!this.observer[type]) {
            this.observer[type] = [];
        }
        this.observer[type].push(fn);
    }
    emit(type) {
        if (!type || !this.observer[type]) {
            return;
        }
        this.observer[type].forEach(fn => fn());
    }
    remove(type) {
        if (!type) {
            return;
        }
        delete this.observer[type];
    }
}
